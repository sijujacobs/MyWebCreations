Date Difference

							
dateDifference_1: function(pDate){
	var dayDifference = 0;
	var today = new Date();
	var podDateParts = pDate.split('/');
	var podDate = new Date(podDateParts[0] + "/" +  podDateParts[1]+ "/" + podDateParts[2]);
	var thisDate = new Date( today.getMonth() + "/" + today.getDate() + "/" + today.getFullYear());
	dayDifference = (thisDate - podDate) / 1000 / 60 / 60 / 24;
	//console.log("Difference between POD date and Today is : " + dayDifference);
	return dayDifference;
}


dateDifference_2: function(pDate){
	var podDateParts = pDate.split('/');
	var newPodDateFormat = podDateParts[2] + "-" + podDateParts[0] + "-" + podDateParts[1];
	var pDateMS=new Date(newPodDateFormat).getTime();
	var todayDateMS = Date.now();
	var dayMS= 1000 * 3600 * 24;
	var timeDiff = todayDateMS - pDateMS;
	var nbrDays = Math.ceil(timeDiff / dayMS)-1;
	return nbrDays;
}

Wait for Elememt

							
waitForElement:function(){
	var thisDocument = this;
	var timer;
	var expectedElement = $('#demoIframe').contents().find('div#content-footer');
	if($(expectedElement).length > 0){
		clearTimeout(timer);
	}else{
		if(thisDocument.WAITCOUNT < 10){
			timer = setTimeout(function(){ thisDocument.waitForElement();}, 500);
			++thisDocument.WAITCOUNT;
			console.log('WaitForElement :: exptedElemLen : ' + $(expectedElement).length);
		}else{
			clearTimeout(timer);
		}
	}
}
